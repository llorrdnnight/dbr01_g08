CREATE TABLE IF NOT EXISTS dbr01_g08.begeleider(
id_begeleider	Integer						NOT NULL,
leeftijd		Integer						NOT NULL,
telefoonnummer	float(53)					NOT NULL,
voornaam		varchar(35)					NOT NULL,
achternaam		varchar(35)					NOT NULL,
rijbewijs		ENUM(a, b, c, d, g, am)		NOT NULL,
ehbo			ENUM(ja, nee)				NOT NULL,
geboortedatum	DATE						NOT NULL,
vooropleiding	TINYTEXT					NOT NULL,
CONSTRAINT begeleider_PK PRIMARY KEY (id_begeleider)
)

CREATE TABLE IF NOT EXISTS dbr01_g08.progamma(
id_progamma		Integer						NOT NULL,
id_begeleider 		Integer						NOT NULL,
naam			VARCHAR(35)					NOT NULL,
duur			VARCHAR(35)					NOT NULL,
CONSTRAINT progamma_PK PRIMARY KEY (id_progamma),
CONSTRAINT begeleider_fk FOREIGN KEY (id_begeleider)
)
